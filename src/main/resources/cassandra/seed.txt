CREATE TABLE ticks(
  symbol TEXT,
  timeframe TEXT,
  tick_time TIMESTAMP,
  created TIMESTAMP,
  open_price DOUBLE,
  high_price DOUBLE,
  low_price DOUBLE,
  close_price DOUBLE,
  volume BIGINT,
  sma6 DOUBLE,
  sma10 DOUBLE,
  sma20 DOUBLE,
  sma50 DOUBLE,
  sma100 DOUBLE,
  sma200 DOUBLE,
  dv_buying DOUBLE,
  dv_buying_sma20 DOUBLE,
  dv_selling DOUBLE,
  dv_selling_sma20 DOUBLE,
  rti3 DOUBLE,
  rti4 DOUBLE,
  rti5 DOUBLE,
  rti6 DOUBLE,
  three_inside_days DOUBLE,
  ADR DOUBLE,
  ATR20 DOUBLE,
  market_cap DOUBLE,
  shares_outstanding BIGINT,
  shares_float BIGINT,
  eps_last_qtr DOUBLE,
  iwm_close_price DOUBLE,
  iwm_sma10 DOUBLE,
  iwm_sma20 DOUBLE,
  iwm_sma50 DOUBLE,
  iwm_di_positive DOUBLE,
  iwm_di_negative DOUBLE,
  PRIMARY KEY (symbol, timeframe, tick_time)
) WITH CLUSTERING ORDER BY (timeframe ASC, tick_time DESC);

CREATE TABLE pattern_match (
    symbol TEXT,
	pattern_name TEXT,
	pattern_time timestamp,
    "created" timestamp,
	PRIMARY KEY(symbol, pattern_name, pattern_time)
) WITH CLUSTERING ORDER BY (pattern_name ASC, pattern_time DESC);
